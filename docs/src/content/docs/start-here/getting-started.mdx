## Install

<Tabs>
  <TabItem label="npm">
    ```bash
      # Using NPM
      npm install astro-breadcrumbs
    ```

  </TabItem>
  <TabItem label="pnpm">
    ```bash
      # Using PNPM
      pnpm add astro-breadcrumbs
    ```

  </TabItem>
  <TabItem label="yarn">
    ```bash
      # Using Yarn
      yarn add astro-breadcrumbs
    ```

  </TabItem>
</Tabs>

## Usage

```astro
---
import { Breadcrumbs } from "astro-breadcrumbs";

import "astro-breadcrumbs/breadcrumbs.css";

// or import the scss file
// import "astro-breadcrumbs/breadcrumbs.scss";
---

<Breadcrumbs />
```
